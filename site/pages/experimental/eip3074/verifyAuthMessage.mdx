# verifyAuthMessage

Verifies if a signed auth message was generated by the provided address.

## Usage

:::code-group

```ts twoslash [example.ts]
import { verifyAuthMessage } from 'viem/experimental'
import { account, walletClient } from './config'

const args = {
  chainId: 1,
  commit: keccak256('0x1234'),
  invokerAddress: '0xd2135CfB216b74109775236E36d4b433F1DF507B',
  nonce: 69,
}

const signature = await walletClient.signAuthMessage({
  ...args,
  account,
})

const valid = await verifyAuthMessage({ // [!code focus:99]
  ...args,
  address: account.address,
  signature,
})
```

```ts twoslash [config.ts] filename="config.ts"
import { createWalletClient, http } from 'viem'
import { mainnet } from 'viem/chains'
import { walletActionsEip3074 } from 'viem/experimental'
 
export const walletClient = createWalletClient({
  chain: mainnet,
  transport: http(),
}).extend(walletActionsEip3074())
```

:::

## Returns

`boolean`

Whether the provided `address` generated the `signature`.

## Parameters

### address

- **Type:** [`Address`](/docs/glossary/types#address)

The Ethereum address that signed the original message.

```ts

  address: '0xf39fd6e51aad88f6f4ce6ab8827279cfffb92266', // [!code focus:1]
  chainId: 1,
  commit: keccak256('0x1234'),
  invokerAddress: '0xd2135CfB216b74109775236E36d4b433F1DF507B',
  nonce: 69,
  signature:
    '0x66edc32e2ab001213321ab7d959a2207fcef5190cc9abb6da5b0d2a8a9af2d4d2b0700e2c317c4106f337fd934fbbb0bf62efc8811a78603b33a8265d3b8f8cb1c',
})
```
